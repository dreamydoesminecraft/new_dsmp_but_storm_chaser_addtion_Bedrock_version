# Wind Advisory
execute if score global nws_alert matches 20 run title @a title {"text":"WIND ADVISORY","color":"gray","bold":true}
execute if score global nws_alert matches 20 run title @a subtitle {"text":"Strong winds expected","color":"white"}

# Flood Advisory
