import { world, system } from "@minecraft/server";

system.runInterval(() => {
  for (const vehicle of world.getEntities({ tags: ["vehicle"] })) {
    const speed = vehicle.getProperty("vehicle_speed") ?? 0;

    // Move forward
    const forward = vehicle.getViewDirection();
    vehicle.teleport(
      {
        x: vehicle.location.x + forward.x * speed,
        y: vehicle.location.y,
        z: vehicle.location.z + forward.z * speed
      },
      vehicle.dimension,
      vehicle.rotation.x,
      vehicle.rotation.y
    );

    // Damage effects
    const damage = vehicle.getProperty("vehicle_damage") ?? 0;
    if (damage >= 50) vehicle.kill();
  }
}, 2);
